94 // Verify that xrif difference/un-difference cycle using pixels works with white noise
95 // For int16_t
96 START_TEST (ascii_pixel_none_lz4_xrif_h)
97 {
98  xrif_t hand = NULL;
99 
100  xrif_error_t rv = xrif_new(&hand);
101 
102  ck_assert( hand != NULL);
103  ck_assert( rv == XRIF_NOERROR );
104 
105  char fname[] = "../src/xrif.h";
106 
107  FILE * fp = fopen(fname ,"rb");
108  //get size.
109  struct stat stbuf;
110 
111  if ((fstat(fileno(fp), &stbuf) != 0) || (!S_ISREG(stbuf.st_mode)))
112  {
113  ck_assert( 0 );
114  }
115 
116  size_t sz = stbuf.st_size;
117 
118  rv = xrif_configure(hand, XRIF_DIFFERENCE_PIXEL, XRIF_REORDER_NONE, XRIF_COMPRESS_LZ4);
119  ck_assert( rv == XRIF_NOERROR );
120 
121  rv = xrif_set_size(hand, sz, 1, 1, 1, XRIF_TYPECODE_INT8);
122  ck_assert( rv == XRIF_NOERROR );
123 
124  rv = xrif_allocate(hand);
125  ck_assert( rv == XRIF_NOERROR );
126 
127  rv = fread(hand->raw_buffer, 1, sz, fp);
128  ck_assert( rv == sz );
129 
130  int8_t * compbuff = (int8_t *) malloc( hand->width*hand->height*hand->frames*sizeof(int8_t));
131  memcpy(compbuff, hand->raw_buffer, hand->width*hand->height*hand->frames*sizeof(int8_t));
132 
133  xrif_encode(hand);
134  xrif_decode(hand);
135 
136  int neq = 0;
137  for( size_t i = 0 ; i < hand->width*hand->height*hand->frames ; ++i )
138  {
139  if(hand->raw_buffer[i] != compbuff[i])
140  {
141  ++neq;
142  }
143  }
144  ck_assert( neq == 0 );
145 
146  free(compbuff);
147  double dt_difference = xrif_difference_time(hand);
148  double dt_reorder = xrif_reorder_time(hand);
149  double dt_compress = xrif_compress_time(hand);
150  double dt_encode = xrif_encode_time(hand);
151  double dt_ratio = xrif_compression_ratio(hand);
152  double dt_decompress = xrif_decompress_time(hand);
153  double dt_unreorder = xrif_unreorder_time(hand);
154  double dt_undifference = xrif_undifference_time(hand);
155  double dt_decode = xrif_decode_time(hand);
156 
157  printf(" ratio: %0.2f %%\n", dt_ratio/1.0*100);
158  printf(" encode: %0.2f MB/s\n", sz/(dt_encode/1.0)/1048576);
159  printf(" difference: %0.2f MB/s\n", sz/(dt_difference/1.0)/1048576);
160  printf(" reorder: %0.2f MB/s\n", sz/(dt_reorder/1.0)/1048576);
161  printf(" compress: %0.2f MB/s\n", sz/(dt_compress/1.0)/1048576);
162  printf(" decode: %0.2f MB/s\n", sz/(dt_decode/1.0)/1048576);
163  printf(" decompress: %0.2f MB/s\n", sz/(dt_decompress/1.0)/1048576);
164  printf(" unreorder: %0.2f MB/s\n", sz/(dt_unreorder/1.0)/1048576);
165  printf(" undifference: %0.2f MB/s\n", sz/(dt_undifference/1.0)/1048576);
166 
167  rv = xrif_delete(hand);
168  ck_assert( rv == XRIF_NOERROR );
169 }
170 END_TEST
