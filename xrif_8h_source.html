229 /** \defgroup xrif_interface Top-level interface
230  * The top-level interface to the XRIF library.
231  *
232  * The following code illustrates how the XRIF library shoud used to compress data under most circumstances.
233  * Lower level access is provided if fine-grain control is needed.
234  *
235  * First, create the xrif library handle:
236  * \code
237  * xrif_error_t rv;
238  * xrif_t handle; //note: this is a pointer type
239  * rv = xrif_new(&handle); //note: the argument is a pointer to pointer
240  * //check rv for errors here . . .
241  * \endcode
242  * You now have an initialized xrif handle. The `xrif_t` type can not be used until `xrif_new` has been called on it.
243  * Never call `xrif_new` twice on the same handle, unless `xrif_delete` (see below) has been called.
244  *
245  * Next you need to configure and allocate the handle:
246  * \code
247  * rv = xrif_set_size(handle, 64,64, 1,1024, XRIF_TYPECODE_INT16);
248  * rv = xrif_configure(handle, XRIF_DIFFERENCE_PREVIOUS, XRIF_REORDER_BYTEPACK, XRIF_COMPRESS_LZ4); //Note this is optional if defaults are ok.
249  * rv = xrif_allocate(handle);
250  * \endcode
251  * The above results in an xrif handle prepared to operate on 1024 signed 16-bit images of size 64x64. (Note that you should check the value of rv for errors!)
252  *
253  * Now to compress data:
254  * \code
255  * memcpy(xrif->raw_buffer, my_source, handle->width*handle->height*handle->depth*handle->frames*handle->data_size); //You are responsible for `my_source`.
256  * rv = xrif_encode(handle);
257  * printf("%f\%\n", handle->compression_ratio*100.0);
258  * \endcode
259  * After this `handle->raw_buffer` contains the compressed data. This is because `handle->compress_on_raw == 1` by default, causing the `raw_buffer` to be re-used.
260  * This also illustrates the built in performance monitoring, which is on by default but can be turned off.
261  *
262  * To decompress:
263  * \code
264  * rv = xrif_decode(handle);
265  * memcpy(my_dest, xrif->raw_buffer, xrif->width*xrif->height*xrif->depth*xrif->frames*xrif->data_size); //You are responsible for `my_dest`.
266  * \endcode
267  * after which `my_dest` will contain the original data.
268  *
269  * To start over, use:
270  * \code
271  * rv = xrif_reset(handle);
272  * \endcode
273  *
274  * And when you are done with the handle, it can be fully de-allocated with
275  * \code
276  * rv = xrif_delete(handle);
277  * \endcode
278  * after which the handle should not be used again, unless `xrif_new` is called on it.
279  * @{
280  */
